    ProductID INT,
    Price DECIMAL(10, 2),
    Weight FLOAT,
    ProductCode CHAR(5),
    ProductName VARCHAR(100),
    Description TEXT,
    ProductImage BLOB,
    LargeImage LONGBLOB,
    Category ENUM('Electronics', 'Clothing', 'Furniture'),
    MfgDate DATE,
    MfgTime TIME,
    MfgYear YEAR
